\begin{frame}[allowframebreaks]{Miért van szükség az operációs rendszerre?}
	
Két nézõpont: felhasználó -- programozó
		
		\begin{enumerate}
			\item A felhasználó szempontjából
			
			\begin{itemize}
				\item a hardver kezelése nagyon bonyolult:
				\begin{itemize}
					\item csak egyszerû parancsokat ért meg: pörgesd fel a lemezt, vidd az olvasófejet egy adott szektor fölé, olvass be x bájtot stb.
					\item mit szeretne a felhasználó: játszd le a behelyezett lemezen lévõ filmet
					\item ezt teszi lehetõvé az operációs rendszer
				\end{itemize}

				\item sokféle hardvereszközt kell kezelni:
				
				\begin{itemize}
					\item fájlok tárolására szolgáló eszközök: merevlemez, hajlékonylemez, CD, DVD, pendrive, memóriakártya, hálózati meghajtók
					\item különbözõ fajta nyomtatók: mátrix, tintasugaras, lézer, 
					\item mi lenne ha ezeket különbözõképpen kellene kezelni?
				\end{itemize}
				
			\end{itemize}
			\framebreak
			\item A programozó szempontjából
			
			\begin{itemize}
				\item ugyanezek a szempontok a programozó szempontjából is felmerülnek
				\item egyszer mindenképpen meg kell írni ezeket a hardverkezelõ programrészleteket, de nem lenne jó, ha
				
				\begin{itemize}
					\item[\dots] minden programozónak értenie kellene az alacsonyszintû programozáshoz
					\item[\dots] minden programozónak ismernie kellene a különbözõ gyártók eszközeinek programozását
					\item[\dots] egy új eszköz megjelenésekor újra kellene írni minden programot
					\item[\dots] minden programozó hozzáférne a hardvereszközök alacsonyszintû kezeléséhez, nagyon veszélyes lenne!
					\item[\dots] ha az egyes programozóknak kellene összehangolniuk az eszközök mûködését és osztozkodniuk a hardver erõforrásokon
				\end{itemize}
				
			\end{itemize}
			
		\end{enumerate}
		
\end{frame}

\begin{frame}[allowframebreaks]{A megoldás}

\begin{itemize}
	\item Minden részfeladatot ott oldjunk meg, ahol az a legmegfelelõbb:
	
	\begin{itemize}
		\item a hardvereszközök kezelését írja meg az eszköz gyártója, merevlemezgyártó, CD olvasó gyártója, nyomtatógyártó, stb.
		\item mindenki tartson be bizonyos megállapodásokat, szabványokat
		\item mindenki felhasználhatja a többi gyártó szabványos megoldásait
		\item az eszközök együttmûködését biztosító programokat írja meg egyetlen szoftvergyártó (operációs rendszer)
		\item a többi szoftvergyártó használja az operációs rendszer szolgáltatásait, ne foglalkozzon az alatta lévõ megoldásokkal
	\end{itemize}
	
	\item Ezt nevezzük réteges felépítésnek
	\item Ezt a megoldást még nagyon sokszor fogjuk látni különbözõ informatika rendszerekben, a legkülönbözõbb tantárgyakban
	\item Minden rendszer ilyen?
\end{itemize}
	
\end{frame}

\begin{frame}[allowframebreaks]{Az operációs rendszer fogalma}
	
	\begin{itemize}
		\item A réteges felépítés középsõ rétegének (RENDSZERPROGRAMOK) egy részét nevezzük operációs rendszernek
		\item A határok nem pontosan definiáltak
		\item Az egyértelmû, hogy a rendszermag (kernel) mindig az operációs rendszer része
		\item A kernel irányítja a számítógép mûködését 
		\item Innen ered az elnevezés:
		
		\begin{itemize}
			\item \textbf{Operating System}, magyarul \textbf{Mûködtetõ rendszer}
			\item Az Operációs rendszer egy tükörfordítás, nem igazán szerencsés
			\item A késõbbiekben röviden \textbf{OS}-nek is fogjuk nevezni az angol elnevezés rövidítéseként
		\end{itemize}
		
		\item Van, aki az operációs rendszer részének tekinti a gép használatát megkönnyítõ segédprogramokat is
		\item Ebben a tárgyban elsõsorban a kernellel fogunk foglalkozni
		
	\end{itemize}
	
\end{frame}

\subsection{Az operációs rendszer feladata}

Az operációs rendszer alapvetõen két feladatot lát el:

\begin{enumerate}
	\item A \textit{\textbf{felhasználó}} számára kényelmes és hatékony \textit{\textbf{munkavégzést}} biztosít a számítógépen
	\item Biztosítja a számítógép \textit{\textbf{erõforrásainak}} hatékony, \textit{\textbf{biztonságos}} és \textit{\textbf{igazságos}} használatát (irányítás, koordinálás)

	A \textbf{\textit{dõlt, félkövér}} betûvel írt fogalmak további elemzést igényelnek. Errõl (is) fog szólni ez a félév.

	\end{itemize}

	\item a tranzisztorok megjelenésével megnõtt a számítógépek megbízhatósága
	\item a számítógép hasznos munkát tudott végezni
	\item kezelõszemélyzet: tervezõk, gyártók, kezelõk, programozók, karbantartók
	\item nagyszámítógépek, mainframe-ek, légkondicionált gépterem
	\item A munkafolyamat:
	
	\begin{itemize}
		\item programozó megírja a programot papíron, FORTRAN, vagy assembly nyelven
		\item lyukkártyára lyukasztják
		\item a kártyacsomagot beviszik a fogadó terembe, átadják a kezelõknek
		\item a kezelõ lefuttatja a programot, kinyomtatja az eredményt
		\item a kiviteli teremben a programozó megkapja a kinyomtatott eredményt (hibalistát)
	\end{itemize}
	
	\item Idõveszteség: ha egy feladathoz szükség van valamilyen eszközre, pl. FORTRAN fordító, akkor a kezelõ betölti azt is a gépbe
	\item Cél: minél jobban használjuk ki a rendelkezésre álló gépidõt
	
	\framebreak
	
	\item Megoldás: \textbf{kötegelt feldolgozás}, \textbf{batch feldolgozás}
	
	\begin{itemize}
		\item a fogadóteremben összeválogatják a hasonló feladatokat (job), pl. a FORTRAN fordítót igénylõket
		\item ezeket egymás után futtatják
		\item csak egyszer kell betölteni a FORTRAN fordítót
	\end{itemize}
	
	\framebreak
	
	\item További hatékonyságnövelõ megoldások:
	
	\begin{itemize}
		\item mágnesszalag használata
		\item beállítunk egy vagy két olcsó számítógépet a periféria mûveletekre
		\item online helyett offline mûködés
		\item egyszerû monitor program (resident monitor) (az elsõ operációs rendszer)
		\item jobvezérlõ program
		\item memória felosztása: monitor memóriaterület -- felhasználói terület
	\end{itemize}
	
	\framebreak
	
	\item Új munkafolyamat:
	
	\begin{itemize}
		\item a fogadó oldalra beállítunk egy kisebb számítógépet, ami beolvassa a lyukkártyákat és mágnesszalagra írja azokat
		\item a mágnesszalagot az operátorok átviszik a nagy számítógéphez
		\item a monitor program beolvassa a mágnesszalagot és feldolgozza a rajta lévõ feladatokat a jobvezérlõ programok alapján
		\item a programok eredményeit felírja egy másik mágnesszalagra
		\item az operátorok kiviszik a mágnesszalagot a kimeneti teremben lévõ harmadik számítógéphez
		\item a kimeneti számítógép kinyomtatja a programok eredményeit
	\end{itemize}
	
	\framebreak
	
	\item Eredmény:
	
	\begin{itemize}
		\item a drága központi számítógép ideje nagy részében a feladatokkal foglalkozhat
		\item a lassú periféria mûveleteket olcsóbb számítógépekkel végeztetjük
		\item az operátorok munkájának egy részét (programok betöltése, elindítása) az állandóan a memóriában lévõ monitor program végzi, ez is gyorsítja a központi számítógép munkáját, mert nem kell várni az operátorra
	\end{itemize}
	
	\item A mai számítógépekben használjuk ezeket a módszereket?
	
	\framebreak
	
	\item Hol használjuk ezeket a módszereket a mai számítógépekben?
	
	\begin{itemize}
		\item Nyomtatás: 
		
		\begin{itemize}
			\item Fájl menü, Nyomtatás parancs
			\item Dolgozhatunk tovább
			\item A program a háttérben elküldi a nyomtatnivalót a nyomtatásvezérlõnek, és az elvégzi a nyomtatást
			\item A nyomtatóban is van egy processzor, ami elvégzi a nyomtatást
		\end{itemize}
		
		\item Nagyfelbontású 2D/3D grafikus megjelenítés:
		
		\begin{itemize}
			\item a mai korszerû grafikus kártyákon van processzor és memória
			\item a CPU elküldi a kirajzolandó kép adatait a grafikus kártyának és dolgozik tovább
			\item a grafikus kártyán lévõ processzor rajzolja ki a képet
		\end{itemize}
		
		\item Batch fájlok DOS és Windows alatt (*.BAT), script fájlok Linux alatt
		
		\begin{itemize}
			\item Egy szöveg fájlban leírjuk azokat a parancsokat, amelyeket egymás után végre akarunk hajtani
			\item A batch/script fájl elindításakor az operációs rendszer sorban végrehajtja a benne szereplõ parancsokat
			\item Használhatunk vezérlõszerkezeteket (elágazás, ciklus), lásd Programozási alapok 1.
		\end{itemize}
				
	\end{itemize}
		
\end{itemize}

\subsection{3. generáció}
\label{sec:3Generacio}

\begin{frame}[allowframebreaks]{3. generáció (1965-1980): integrált áramkörök}
	
	\begin{itemize}
		\item A 2. generációban alkalmazott módszerek megoldottak egy fontos problémát: megnövelték a drága számítógépek kihasználtságát, ezzel lecsökkentették a fajlagos költségeket
		\item Viszont szültek egy új problémát: a számítóközpontokban kétféle számítógép típus jelent meg
		
		\begin{enumerate}
			\item nagy teljesítményû tudományos komputerek
			\item kisebb teljesítményû perifériakezelõ számítógépek
		\end{enumerate}
		
		\item Ezeket eltérõ módon kellett programozni, több, eltérõ tudású programozóra, fejlesztõeszközre volt szükség
		\item Ez megnövelte a költségeket
		\item A számítógépgyártók erre a problémára kerestek megoldást
		\framebreak
		\item Az IBM válasza: System/360 gépcsalád
		
		\begin{itemize}
			\item a család különbözõ méretû (memória, gyorsaság, perifériák, ár, teljesítmény) számítógépekbõl állt
			\item de ezek egymással kompatibilisek voltak, azaz bármelyik gépre írt program fut az összes többin is (legalábbis elvileg)
			\item ez sok elõnnyel járt, de sokkal bonyolultabb operációs rendszerre volt szükség
			\item Új perifériák: mágnesdob, mágneslemez
			
			\begin{itemize}
				\item nagykapacitású
				\item gyors
				\item véletlen hozzáférésû (random access): a mágnesszalaggal ellentétben nem csak sorban egymás után lehet elérni a tárolt adatokat, hanem tetszõlegesen (így mûködnek a mai merevlemezek és a legtöbb háttértár, kivéve a szalagos egység -- streamer)
			\end{itemize}
			
		\end{itemize}
		
		\framebreak
		
		\item Új technikák
		
		\begin{itemize}
			\item A véletlen elérésû tárolók két új technikai megoldás kidolgozását tették lehetõvé:
			
			\begin{itemize}
				\item spooling technika
				\item multiprogramozás
			\end{itemize}
			
			\item Ezeket ma is folyamatosan használjuk
			
		\end{itemize}
		
		\framebreak
		
		\item Spooling technika
		
		\begin{itemize}
			\item simultaneous peripherial operation on-line -- on-line szimultán periféria  (háttértárolás)		
			\item a számítógép háttérben el tudja végezni periféria mûveleteket
			\item nincsen szükség külön számítógépre a perifériák kezeléséhez
		\end{itemize}
		
		%\framebreak
		
		\item Multiprogramozás
		
		\begin{itemize}
			\item A spooling technika segítségével a számítógép képes a háttérben végrehajtani a periféria mûveleteket, eközben a feladatok folyhatnak tovább
			\item Ha azonban egyetlen feladat fut a gépen, akkor a periféria mûveletek alatt a számítógép várakozni kényszerül, ezzel lecsökken a számítógép kihasználtsága
			\item A mágneslemezek lehetõvé teszik, hogy a processzor egyszerre több feladat adataihoz férjen hozzá és kiírja azok eredményét
			\item Amikor egy feladat várakozni kényszerül, akkor a processszor elindít egy másik feladatot
			\item Ehhez arra is szükség van, hogy a memóriát továbbtagoljuk, a monitor és az egyes feladatok által használt területekre
			\item Ezek a memóriaterületek egymástól függetlenek, az egyes programok nem férhetnek hozzá a másik program által használt memóriaterülethez
		\end{itemize}
		
		\item Új operációs rendszer (OS/360)
		
		\begin{itemize}
			\item Ezek az új technikák új, a korábbiaknál sokkal nagyobb méretû, összetett operációs rendszert igényeltek
			\item Az IBM ezt OS/360-nak nevezte el
			\item A nagy méretû kód nagyon sok hibát eredményezett, a sok javítás pedig a régi hibák kijavítása mellett újabb hibákat okozott, a hibák száma körülbelül állandó volt
			\item Ennek ellenére ezek a rendszerek elég jól kiszolgálták a felhasználók igényeit
			\item A programozók azonban elkezdték hiányolni a géppel való közvetlen kapcsolatot, ami utoljára az 1. generációs gépeknél volt meg
			\item Erre az igényre adott választ az idõosztás (timesharing) technika
		\end{itemize}
		
		%\framebreak
		
		\item Idõosztás (timesharing)
		
		\begin{itemize}
			\item Minden felhasználó kap egy on-line terminált (képernyõ + billentyûzet), amivel közvetlenül kapcsolódik a számítógéphez
			\item Parancsokat tud begépelni a billentyûzeten, a választ a képernyõn látja
			\item Amikor a processzor felszabadul az éppen ellátott feladat alól, akkor kiválasztja a soron következõt és végrehajtja, az eredményt visszaküldi a felhasználó képernyõjére
			\item Amikor egy felhasználó gondolkodik, vagy éppen gépeli a parancsot, akkor nem terheli a számítógépet, így egyetlen gép sok felhasználót tud interaktívan kiszolgálni, miközben egy nagy kötegelt feladaton is dolgozik
			\item Egy felhasználó viszont képes hosszú idõre lefoglalni a teljes gépet egy sokáig futó parancs kiadásával
			\item Ezek a feladatok tovább bonyolították az operációs rendszert
			\item A legismertebb ilyen operációs rendszer a MULTICS (MULTiplexed Information and Computing Service)
			\item Az utolsó nagy MULTICS rendszert 2000. októberében állították le a Kanadai Védelmi Minisztériumban
		\end{itemize}
				
	\end{itemize}

\subsection{4. generáció}
\label{sec:4Generacio}

\begin{frame}[allowframebreaks]{4. generáció (1980-napjainkig): személyi számítógépek}
	
	\begin{itemize}
		\item A hardvereszközök árának rohamos csökkenésével alapvetõen új irányt vet a számítástechnika fejlõdése:
		
		\begin{enumerate}
			\item Eddig a felhasználó osztozott a számítógépen sok más felhasználóval
			\item Ugyanakkor sok szakember állt rendelkezésre a számítógép üzemeltetéséhez
		\end{enumerate}
		
		\item Amikor megjelent a személyi számítógép (Personal Computer, PC), mindkét irányból jelentõs változás következett be:
		
		\begin{enumerate}
			\item A felhasználó egyedül birtokolja a számítógépet, nem osztozik rajta senkivel
			\item A felhasználónak egyedül kell tudnia kezelni a számítógépet, õ az operátor, a kezelõ, a programozó és a felhasználó egy személyben
		\end{enumerate}
		
		\item Ehhez gyökeresen új megoldásokra, és új operációs rendszerekre volt szükség
		\item Megjelent egy új, eddig ismeretlen szempont az operációs rendszerek fejlesztésében, a felhasználóbarátság
		
		\begin{itemize}
			\item nem elég jónak lennie az operációs rendszernek
			\item az egyszerû felhasználónak meg kell értenie, tudnia kell kezelni, sõt meg kell szeretnie az operációs rendszert, különben másikat választ, vagy távoltartja magát a számítógéptõl
			\item kialakult a piaci verseny fogalma
			\item A személyi számítógépeken mûködõ operációs rendszerek: CP/M, DOS (DR-DOS, MS-DOS, Apple DOS), Mac OS, Windows, Linux
		\end{itemize}
		
		\item Kezdetben az emberek ,,megelégedtek'' a saját személyi számítógéppel
		\item Hamarosan megjelent az igény, hogy a személyi számítógéppel kapcsolódni tudjunk más számítógépekhez, miért?
		
		\begin{itemize}
			\item elektronikus kommunikáció
			\item erõforrások megosztása: saját erõforrások megosztása másokkal, távoli erõforrások elérése
		\end{itemize}
		
		\item Ezen igények kielégítésére megjelentek a számítógépes hálózatok
		\item A hálózatok megjelenése ismét elõtérbe hozott eddig háttérbe szorított szempontokat: biztonság, felhasználók azonosítása, védelme egymástól, jogosultságok kezelése stb.
		\item Ezeket az igényeket kétféle operációs rendszer típussal tudjuk kielégíteni:
		
		\begin{itemize}
			\item hálózati operációs rendszerek:
			
			\begin{itemize}
				\item több felhasználó használhatja ugyanazt a számítógépet
				\item egy felhasználó használhat több különbözõ számítógépet távolról belépve a számítógépre
				\item ezek az operációs rendszerek nem térnek el gyökeresen a korábbi egyfelhasználós rendszerektõl, csak új funkciókat építettek beléjük az új feladatok kezelésére
			\end{itemize}
			
			\item elosztott operációs rendszerek:
			
			\begin{itemize}
				\item Olyan operációs rendszer, ami a felhasználó felé egy hagyományos egyprocesszoros rendszert mutat, miközben több processzor, vagy több teljes számítógép van a rendszerben
				\item A felhasználó nem tudja, hogy ténylegesen melyik gépen, vagy melyik processzoron fut a programja, vagy hol tárolódnak az adatai
				\item Mindezt a rendszer dönti el valamilyen szempontból optimálisan
			\end{itemize}
			
			\item Úgy is megfogalmazhatjuk a különbséget, hogy a hálózati operációs rendszer felhasználója tudja, hogy éppen melyik számítógépen dolgozik, az elosztott rendszer használója nem tudja
			\item Ezt úgy mondjuk, hogy az elosztott rendszer transzparens (átlátszó)
			
		\end{itemize}
				
	\end{itemize}
